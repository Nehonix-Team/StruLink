Quick Start

URL Validation
import { StruLink as __strl__ } from "strulink";

const result = await __strl__.asyncCheckUrl(
  "https://example.com"
);

console.log(result.isValid); // true
Auto-Decode
const decoded = __strl__.autoDetectAndDecode(
  "SGVsbG8gV29ybGQ="
);

console.log(decoded.val()); 
// "Hello World"
Security Analysis
const analysis = __strl__.detectMaliciousPatterns(
  "?user=admin' OR '1'='1"
);

console.log(analysis.isMalicious); 
// true
Encoding
const encoded = __strl__.encode(
  "Hello World",
  "base64"
);

console.log(encoded); 
// "SGVsbG8gV29ybGQ="
